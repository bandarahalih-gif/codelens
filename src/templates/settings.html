🔌 CI/CD Integration

Add CodeLens to your CI pipeline to automatically review code on every push:

# .github/workflows/review.yml
name: Code Review
on: [push, pull_request]
jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Run CodeLens
        run: |
          curl -X POST ${{ secrets.CODELENS_URL }}/api/v1/review \
            -F "file=@src/main.py" \
            -H "Authorization: Bearer ${{ secrets.CODELENS_API_KEY }}"
